### 2.2 SIWE (Sign-In with Ethereum) 实现

```go
// SIWE 消息格式
type SIWEMessage struct {
    Domain       string    `json:"domain"`        // 域名
    Address      string    `json:"address"`       // 钱包地址
    Statement    string    `json:"statement"`     // 声明
    URI          string    `json:"uri"`           // URI
    Version      string    `json:"version"`       // "1"
    ChainID      int       `json:"chain_id"`      // 56 (BSC)
    Nonce        string    `json:"nonce"`         // 随机数
    IssuedAt     time.Time `json:"issued_at"`     // 签发时间
    ExpirationAt time.Time `json:"expiration_at"` // 过期时间
    RequestID    *string   `json:"request_id"`    // 可选请求 ID
}

// 签名验证
func VerifySIWE(message SIWEMessage, signature string) (bool, error) {
    // 1. 构建签名消息
    formattedMessage := fmt.Sprintf(
        "%s wants you to sign in with your Ethereum account:\n%s\n\n%s\n\nURI: %s\nVersion: %s\nChain ID: %d\nNonce: %s\nIssued At: %s",
        message.Domain,
        message.Address,
        message.Statement,
        message.URI,
        message.Version,
        message.ChainID,
        message.Nonce,
        message.IssuedAt.Format(time.RFC3339),
    )
    
    // 2. 恢复公钥
    hash := crypto.Keccak256Hash([]byte(formattedMessage))
    prefixedHash := accounts.TextHash(hash.Bytes())
    
    pubKey, err := crypto.SigToPub(prefixedHash, common.FromHex(signature))
    if err != nil {
        return false, err
    }
    
    // 3. 验证签名地址
    recoveredAddress := crypto.PubkeyToAddress(*pubKey)
    expectedAddress := common.HexToAddress(message.Address)
    
    return strings.EqualFold(recoveredAddress.Hex(), expectedAddress.Hex()), nil
}

// 安全措施
// 1. Nonce 一次性使用，防止重放攻击
// 2. 设置合理的过期时间 (15 分钟)
// 3. 验证 Domain 和 URI 匹配
// 4. 记录登录日志用于审计
```

```go
func (s *EncryptionService) Encrypt(plaintext string) (string, error) {
    input := &kms.EncryptInput{
        KeyId:       aws.String(s.keyID),
        Plaintext:   []byte(plaintext),
        EncryptionContext: map[string]string{
            "purpose": "field-encryption",
            "service": "agarvibe",
        },
    }
    
    result, err := s.kmsClient.Encrypt(context.Background(), input)
    if err != nil {
        return "", err
    }
    
    return base64.StdEncoding.EncodeToString(result.CiphertextBlob), nil
}

// 解密字段
func (s *EncryptionService) Decrypt(ciphertext string) (string, error) {
    data, err := base64.StdEncoding.DecodeString(ciphertext)
    if err != nil {
        return "", err
    }
    
    input := &kms.DecryptInput{
        KeyId: aws.String(s.keyID),
        CiphertextBlob: data,
    }
    
    result, err := s.kmsClient.Decrypt(context.Background(), input)
    if err != nil {
        return "", err
    }
    
    return string(result.Plaintext), nil
}
```

### 5.3 CSRF 防护

```go
// CSRF Token 生成
func GenerateCSRFToken(sessionID string) string {
    token := make([]byte, 32)
    rand.Read(token)
    
    // 绑定 Session
    hmac := hmac.New(sha256.New, []byte(sessionID))
    hmac.Write(token)
    signature := hmac.Sum(nil)
    
    return fmt.Sprintf("%s.%s", 
        base64.URLEncoding.EncodeToString(token),
        base64.URLEncoding.EncodeToString(signature))
}

// CSRF Token 验证
func ValidateCSRFToken(sessionID, token string) bool {
    parts := strings.Split(token, ".")
    if len(parts) != 2 {
        return false
    }
    
    tokenBytes, _ := base64.URLEncoding.DecodeString(parts[0])
    providedSig, _ := base64.URLEncoding.DecodeString(parts[1])
    
    hmac := hmac.New(sha256.New, []byte(sessionID))
    hmac.Write(tokenBytes)
    expectedSig := hmac.Sum(nil)
    
    return subtle.ConstantTimeCompare(providedSig, expectedSig) == 1
}

// SameSite Cookie 属性
```

```go
// Redis + 滑动窗口限流
type RateLimiter struct {
    redis *redis.Client
}

func (l *RateLimiter) IsAllowed(key string, limit int, window time.Duration) bool {
    now := time.Now().UnixNano()
    windowStart := now - window.Nanoseconds()
    
    pipe := l.redis.Pipeline()
    
    // 移除过期分数
    pipe.ZRemRangeByScore(key, "0", strconv.FormatInt(windowStart, 10))
    
    // 添加当前请求
    pipe.ZAdd(key, redis.Z{Score: float64(now), Member: strconv.FormatInt(now, 10)})
    
    // 设置过期时间
    pipe.Expire(key, window*2)
    
    // 计数
    pipe.ZCard(key)
    
    cmds, _ := pipe.Exec(context.Background())
    count := cmds[len(cmds)-1].(*redis.IntCmd).Val()
    
    return count <= int64(limit)
}

// 限流配置
rateLimits := map[string]RateLimitConfig{
    "auth:login":           {Limit: 5, Window: 1 * time.Minute},   // 登录 5 次/分钟
```

```go
// 智能 IP 封禁
type IPBlocklist struct {
    redis *redis.Client
}

// 自动封禁 (失败次数过多)
func (b *IPBlocklist) RecordFailure(ip string, reason string) {
    key := fmt.Sprintf("ip:failures:%s", ip)
    count := b.redis.Incr(key).Val()
    b.redis.Expire(key, 1*time.Hour)
    
    if count >= 10 {
        b.Block(ip, 24*time.Hour, fmt.Sprintf("Too many %s failures", reason))
    }
}

// 手动封禁 (管理员操作)
func (b *IPBlocklist) Block(ip string, duration time.Duration, reason string) {
    key := fmt.Sprintf("ip:blocked:%s", ip)
    b.redis.Set(key, reason, duration)
    
    // 发布事件，同步到所有节点
    b.redis.Publish(context.Background(), "ip:blocked", map[string]interface{}{
        "ip": ip,
        "reason": reason,
        "expires_at": time.Now().Add(duration),
    })
}

// 检查是否被封禁
func (b *IPBlocklist) IsBlocked(ip string) (bool, string) {
    key := fmt.Sprintf("ip:blocked:%s", ip)
    reason := b.redis.Get(key).Val()
    return reason != "", reason
}
```

```go
}

// 数据可携带权 (导出个人数据)
func (s *GDPRService) ExportUserData(userID string) (map[string]interface{}, error) {
    data := make(map[string]interface{})
    
    // 导出用户资料
    var user User
    s.db.First(&user, userID)
    data["user"] = user
    
    // 导出记忆
    var memories []MemoryIndex
    s.db.Where("user_id = ?", userID).Find(&memories)
    data["memories"] = memories
    
    // 导出订单
    var orders []Order
    s.db.Where("user_id = ?", userID).Find(&orders)
    data["orders"] = orders
    
    // 导出 NFT 持有记录
    var nfts []NFTToken
    s.db.Where("owner_address = ?", user.WalletAddress).Find(&nfts)
    data["nfts"] = nfts
    
    return data, nil
}
```

## 10. 应急响应

### 10.1 安全事件分级

```markdown
## P0 - 严重 (Critical)
- 用户数据大规模泄露 (> 10,000 用户)
- 私钥/主密钥泄露
- 智能合约漏洞导致资金损失
- 响应时间：15 分钟内

## P1 - 高危 (High)
- 认证绕过
- SQL 注入成功
- 支付逻辑漏洞
- 响应时间：1 小时内

## P2 - 中危 (Medium)
- XSS 漏洞
- CSRF 漏洞
- 信息泄露
- 响应时间：24 小时内

## P3 - 低危 (Low)
- 配置不当
- 缺少安全头
- 响应时间：下次发布修复
```

### 10.2 应急响应流程

```markdown
1. 发现与报告
   • 监控告警触发
   • 用户报告
   • 白帽子披露
   
2. 初步评估 (15 分钟)
   • 确认事件真实性
   • 评估影响范围
   • 确定事件级别
   
3. 遏制损害 (1 小时)
   • 隔离受影响系统
   • 撤销泄露凭证
   • 暂停可疑账户
   
4. 根除原因 (4 小时)
   • 修复漏洞
   • 清除恶意代码
   • 更新防火墙规则
   
5. 恢复服务 (2 小时)
   • 验证修复效果
   • 逐步恢复服务
   • 持续监控
   
6. 事后总结 (24 小时)
   • 编写事故报告
   • 制定改进措施
   • 更新应急预案
```
